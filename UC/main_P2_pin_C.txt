#include "stm32f1xx.h"

int main(void)
{

	// Enable clock for GPIOB and GPIOC
	RCC->APB2ENR |= (RCC_APB2ENR_IOPBEN | RCC_APB2ENR_IOPCEN);

	GPIOB->CRH   &= ~(0b1111 << GPIO_CRH_MODE11_Pos);
	GPIOB->CRH   |=  (0b1000 << GPIO_CRH_MODE11_Pos);
	GPIOB->ODR   |=  GPIO_ODR_ODR11;

	GPIOC->CRH   &= ~(0b1111 << GPIO_CRH_MODE13_Pos);
	GPIOC->CRH   |=  (0b0001 << GPIO_CRH_MODE13_Pos);
	GPIOC->ODR   |=  GPIO_ODR_ODR13;

	while (1)
		{
		if (!(GPIOB->IDR & GPIO_IDR_IDR11)){
			GPIOC->ODR   &=  ~GPIO_ODR_ODR13;
			}
		else {
			GPIOC->ODR   |=  GPIO_ODR_ODR13;
			}
		}
}
