#include "stm32f1xx.h"

int main(void)
{
	RCC->APB2ENR |=  RCC_APB2ENR_IOPCEN;		//Enable PortC
	GPIOC->CRH   &= ~(0xF << GPIO_CRH_MODE13_Pos);	//Clear Mode PC13
	GPIOC->CRH   |=  (0x1 << GPIO_CRH_MODE13_Pos);	//PC13 output

	RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
	TIM2->PSC  = 1000-1;
	TIM2->ARR  = 8000-1;
	TIM2->DIER = TIM_DIER_UIE;		//Enable Interrupt
	NVIC_EnableIRQ(TIM2_IRQn);		//Activate TIM2_IRQ
	TIM2->CR1  = TIM_CR1_CEN;		//Start timer2

	while(1)
	{
		//Do nothing (Empty loop)
	}

}

void TIM2_IRQHandler()
{
	TIM2 ->SR  &= ~TIM_SR_UIF;      	//Clear Flag
	GPIOC->ODR ^=  GPIO_ODR_ODR13;  	//Toggle PC13
}
