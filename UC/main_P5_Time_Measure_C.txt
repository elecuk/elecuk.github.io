#include "stm32f1xx.h"

volatile uint32_t counter=0, Yekan, Dahgan, Sadgan, Hezargan;
static const uint8_t Table_7_segment[10];

int main(void)
{
	RCC->APB2ENR |= RCC_APB2ENR_IOPAEN;  // Enable clock for GPIOA
	GPIOA->CRL    = 0x11111111; 
	GPIOA->CRH   |= 0x00001111;
	GPIOA->ODR   |= 0xF000; 

	RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
	TIM2->PSC  = 1000-1;
	TIM2->ARR  = 8000-1;
	TIM2->DIER = TIM_DIER_UIE;		 //Enable Interrupt
	NVIC_EnableIRQ(TIM2_IRQn);		 //Activate TIM2_IRQ
      //NVIC->ISER[0] |= (1 << TIM2_IRQn);
	TIM2->CR1  = TIM_CR1_CEN; 		//Start timer2

	while (1)
	{
		GPIOA->ODR = (((GPIOA->ODR | 0xF00 ) & ~0xFF) | Table_7_segment[Yekan])    & ~(GPIO_ODR_ODR11);
		GPIOA->ODR|= GPIO_ODR_ODR7;
		GPIOA->ODR = (((GPIOA->ODR | 0xF00 ) & ~0xFF) | Table_7_segment[Dahgan])   & ~(GPIO_ODR_ODR10);
		GPIOA->ODR = (((GPIOA->ODR | 0xF00 ) & ~0xFF) | Table_7_segment[Sadgan])   & ~(GPIO_ODR_ODR9);
		GPIOA->ODR = (((GPIOA->ODR | 0xF00 ) & ~0xFF) | Table_7_segment[Hezargan]) & ~(GPIO_ODR_ODR8);
	 }
}

void TIM2_IRQHandler()
{
    TIM2 ->SR  &= ~TIM_SR_UIF;      //Clear Flag
    counter++;

    	volatile uint32_t number = counter;

	Yekan    = number % 10;
	number   = number / 10;
	Dahgan   = number % 10;
	number   = number / 10;
	Sadgan   = number % 10;
	number   = number / 10;
	Hezargan = number % 10;

}

static const uint8_t Table_7_segment[10]=
{
	0x3F, // 0
	0x06, // 1
	0x5B, // 2
	0x4F, // 3
	0x66, // 4
	0x6D, // 5
	0x7D, // 6
	0x07, // 7
	0x7F, // 8
	0x6F, // 9
};
